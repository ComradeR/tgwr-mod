	title = ww1_china.47.t
	desc = ww1_china.47.d
	trigger = {
		has_global_flag = second_constitutional_protection_movement

		if = {
			limit = {
				GXC = { exists = yes }
			}
			original_tag = GXC
		}
		else = {
			any_country = {
				owns_state = 591
			}
		}
	}
	fire_only_once = yes
	immediate = {
		hidden_effect = {
			create_dynamic_country = {
				original_tag = CHI

				set_politics = {
					ruling_party = democratic
					elections_allowed = no
				}

				set_popularities = {
					democratic = 32
					neutrality = 65
					fascism = 0
					communism = 3
				}

				set_capital = 592

				set_country_flag = CHI_kuomintang_flag
				set_cosmetic_tag = CHI_uprising #Kuomintang
			}
			KMT = {
				transfer_state = 591
				transfer_state = 592
				transfer_state = 593
				transfer_state = 594

				if = {
					limit = {
						GXC = { exists = yes }
					}
					inherit_technology = GXC
					else = {
						inherit_technology = CHI
					}
				}	

				create_country_leader = {
					name = "Sun Yat-sen"
					picture = "gfx/leaders/CHI/CHI_sun_wen.dds"
					expire = "1935.1.1"
					ideology = socialism
				}

				if = {
					limit = {
						GXC = { exists = yes }
					}

					GXC = {
						transfer_units_fraction = {
							target = KMT
							size = 0.7
							stockpile_ratio = 0.5
							army_ratio = 0.7
							navy_ratio = 0.7
							air_ratio = 0.7
						}
					}

					declare_war_on = {
						target = GXC
						type = civil_war
					}
				}
			}
		}
	}
	option = {
		name = ww1_china.47.a
		ai_chance = { factor = 100 }
	}
	option = {
		name = ww1_china.47.b
		ai_chance = { factor = 0 }
		trigger = {
			original_tag = GXC
		}
		KMT = {
			change_tag_from = GXC
		}
	}
}

country_event = {
	id = ww1_china.54
	title = ww1_china.54.t
	desc = ww1_china.54.d
	picture = GFX_report_event_japan_army_mountainside

	fire_only_once = yes
	trigger = {
		has_country_leader = {
			name = "Sun Yat-sen"
			ruling_only = yes
		}

		OR = {
			date > 1925.3.12
			AND = {
				original_tag = CHI
				has_global_flag = national_protection_war
				OR = {
					NOT = { has_war_with = NPA }
					AND = {
						has_war_with = NPA
						surrender_progress > 0.5
					}
				}
			}
		}
	}
	mean_time_to_happen = { days = 35 }
	immediate = {
		hidden_effect = {
			kill_country_leader = yes

			create_country_leader = {
				name = "Chiang Kai-shek"
				picture = "gfx/leaders/CHI/CHI_chiang_kai_shek.dds"
				expire = "1935.1.1"
				ideology = oligarchism
			}

			add_stability = -0.05

			if = {
				limit = {
					original_tag = CHI
				}
				remove_ideas = CHI_idea_militarist_plot
				set_global_flag = chinese_warlord_crisis
			}
			set_global_flag = sun_yatsen_dead
		}
	}
	option = {
		name = ww1_china.54.a
		ai_chance = { factor = 100 }
	}
}
