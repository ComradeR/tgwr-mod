country_event = {
	id = ww1_firstbalkan.4
	title = ww1_firstbalkan.4.t
	desc = ww1_firstbalkan.4.d
	picture = GFX_report_event_generic_read_write

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = ww1_firstbalkan.4.a
		ai_chance = { factor = 100 }
		
		SER = {
			declare_war_on = {
				type = take_state_focus 
				target = TUR 
				generator = {
					760 106 759
				}
			}
		}
		hidden_effect = {
			set_global_flag = kis_firstbalkan_war
			if = {
				limit = {SER = {has_completed_focus = SER_balkan_league}}
				news_event = { days = 1 id = ww1_firstbalkan.100 }
			}	
			if = {
				limit = {SER = {has_completed_focus = SER_great_serbia}}
				news_event = { days = 1 id = ww1_firstbalkan.102 }
			}	
			BUL = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			GRE = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			MNT = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			add_opinion_modifier = {
				target = TUR
				modifier = large_decrease
			}
			TUR = {
				add_opinion_modifier = {
					target = SER
					modifier = large_decrease
				}
				if = {
					limit = {
						has_completed_focus = TUR_coup_of_savior_officers
						NOT = {
							has_completed_focus = TUR_raid_on_the_sublime_porte
							has_completed_focus = TUR_protect_democracy
							has_completed_focus = TUR_ittihadi_muhammedi
							has_completed_focus = TUR_istirak_mukavemet
						}
					}
					add_timed_idea = {
						idea = TUR_idea_chaos_balkans
						days = 20
					}
				}
				add_opinion_modifier = {
					target = GRE
					modifier = small_decrease
				}
				add_opinion_modifier = {
					target = MNT
					modifier = small_decrease
				}
				add_opinion_modifier = {
					target = BUL
					modifier = large_decrease
				}
			}
		}

		# Bulgaria
		if = {
			limit = {
				BUL = { has_country_flag = join_balkan_league }
			}
			BUL = {
				add_to_war = {
					targeted_alliance = SER 
					enemy = TUR
				}
			}
		}

		# Greece
		if = {
			limit = {
				GRE = { has_country_flag = join_balkan_league }
			}
			GRE = {
				add_to_war = {
					targeted_alliance = SER 
					enemy = TUR
				}
			}
		}

		# Montenegro
		if = {
			limit = {
				MNT = { has_country_flag = join_balkan_league }
			}
			MNT = {
				add_to_war = {
					targeted_alliance = SER 
					enemy = TUR
				}
			}
		}
	}
}

# SERBIA READY FOR PEACE

country_event = {
	id = ww1_firstbalkan.6
	title = ww1_firstbalkan.6.t
	desc = ww1_firstbalkan.6.d
	picture = GFX_report_event_generic_read_write

	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = ww1_firstbalkan.6.a
		ai_chance = { factor = 100 }	
		SER = {
			every_state = {
				limit = {
					OR = {
						is_core_of = SER
						is_claimed_by = SER
					}
					if = {
						limit = {
							ALB = {
								is_subject_of = TUR
							}
						}
						OR = {
							is_owned_by = TUR
							is_owned_by = ALB
						}
						else = {
							is_owned_by = TUR
						}

					}
				}
				SER = { transfer_state = PREV }
			}
		}
		if = {
			limit = {
				GRE = {
					OR = {
						NOT = { has_war_with = TUR } 
						has_country_flag = not_join_balkan_league
					}
				}
				OR = {
					TUR = {
						owns_state = 803
					}
					ALB = {
						owns_state = 803
						is_subject_of = TUR
					}
				}
			}
			SER = { transfer_state = 803 }
		}
		
		if = {
			limit = {
				GRE = { has_war_with = TUR }
			}
			GRE = {
				every_state = {
					limit = {
						OR = {
							is_core_of = GRE
							is_claimed_by = GRE
						}	
						if = {
							limit = {
								ALB = {
									is_subject_of = TUR
								}
							}
							OR = {
								is_owned_by = TUR
								is_owned_by = ALB
							}
							else = {
								is_owned_by = TUR
							}
	
						}
						if = {
							limit = {
								BUL = { has_war_with = TUR }	
							}
							NOT = {
								is_core_of = BUL
							}
						}	
					}
					GRE = { transfer_state = PREV }
				}
			}	
		}
		if = {
			limit = {
				BUL = { has_war_with = TUR }	
			}
			BUL = {
				every_state = {
					limit = { 
						OR = {
							is_core_of = BUL
							is_claimed_by = BUL 
						}
						is_owned_by = TUR
					}
					BUL = { transfer_state = PREV add_state_core = PREV }
				}
				set_province_name = {
					id = 11774
 					name = GYUMYURDZINA_PROV
				}
				set_country_flag = won_balkan_war
			}
		}
		if = {
			limit = {
				ARM = { has_war_with = TUR is_in_faction_with = SER }
			}
			ARM = {
				every_state = {
					limit = {
						is_core_of = ARM
						is_owned_by = TUR
					}
					ARM = {
						transfer_state = PREV
					}
				}
			}
		}
		if = {
			limit = {
				RUS = { has_war_with = TUR is_in_faction_with = SER }
				ARM = { exists = no}
			}
			RUS = {
				every_state = {
					limit = {
						OR = {
							state = 354
							state = 353
							state = 352
						}
						is_owned_by = TUR
					}
					RUS = {
						transfer_state = PREV
					}
				}
			}
		}
		if = {
			limit = {
				MNT = {	has_war_with = TUR }
			}
			every_state = { 
				limit = { 
					is_core_of = MNT 
					OR = {
						is_owned_by = TUR
						if = {
							limit = {
								ALB = {
									is_subject_of = TUR
								}
							}
							is_owned_by = ALB
						}
					}				
				}
				MNT = { transfer_state = PREV }	
			}
		}
		hidden_effect = {
			every_country = { 
				limit = {
					has_war_with = TUR
					OR = {
						tag = SER
						is_in_faction_with = SER
						tag = MNT
						tag = GRE
						tag = BUL
						tag = RUS
						is_subject_of = RUS		
						tag = ARM
					}								
				}
				white_peace = TUR
				set_truce = {
					target = TUR
					days = 365
				}
			}
			news_event = { days = 1 id = ww1_firstbalkan.101 }		
			set_global_flag = kis_firstbalkanwar_victory
			every_country = {
				limit = { 
					OR = { 
						has_country_flag = join_balkan_league 
						has_country_flag = bulgarian_balkan_league_intervention 
					} 
				}
				add_opinion_modifier = { 
					target = TUR 
					modifier = HOI4TGW_Opinion_hostile_relations_post_balkan_war 
				}
			}		
			every_state = {
				limit = {
					OR = {
						is_owned_by = SER
						is_owned_by = MNT
						is_owned_by = BUL
						is_owned_by = GRE
						is_owned_by = ALB
					}
					NOT = { state = 756  state = 757 state = 200}
				}
				remove_core_of = TUR
			}
			TUR = { 
				if = {
					limit = { has_idea = TUR_balkan_separatism }
					remove_ideas = { TUR_balkan_separatism }
					add_ideas = { idea_turkey_minor_separatism } 
				}
			}
			every_country = {
				limit = { has_idea = SER_idea_balkan_liberation }
				remove_ideas = SER_idea_balkan_liberation
			}
		}
	}
}
